// Function to extract latitude and longitude from the URL or use default function getCoordinatesFromURL() { const urlParams = new URLSearchParams(window.location.search); const lat = parseFloat(urlParams.get("lat")); const lng = parseFloat(urlParams.get("lng")); // If valid coordinates are provided, use them; otherwise, return the default if (!isNaN(lat) && !isNaN(lng)) { return { lat: lat, lng: lng }; } // Default coordinates (your hard-coded values) return { lat: 45.423, lng: -75.700 }; } // Function to extract zoom from the URL or use default function getZoomFromURL() { const urlParams = new URLSearchParams(window.location.search); const zoom = parseFloat(urlParams.get("zoom")); // If valid zoom is provided, use it; otherwise, return the default if (!isNaN(zoom)) { return zoom; } // Default zoom level return 14; } // Initialize the map function initMap() { var mapCenter = getCoordinatesFromURL(); // Use URL coordinates or default var mapZoom = getZoomFromURL(); // Use URL zoom or default map = new google.maps.Map(document.getElementById('map'), { zoom: mapZoom, // Correct zoom setting center: mapCenter, // Correctly set the map center mapTypeId: 'satellite', // Use "roadmap" for detailed street and address labels streetViewControl: true, // Enable Street View for more exploration mapTypeControl: true, // Allow switching map types tilt: 0, // Ensures map stays flat (no tilt) styles: [] // Ensure no suppression of labels or features }); // Send the starting coordinates to the webhook fetch(webhookURL, { method: "POST", mode: "no-cors", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ lat: mapCenter.lat, lng: mapCenter.lng }) }) .then(() => console.log("Starting coordinates sent to webhook")) .catch(error => console.error("Error sending starting coordinates:", error)); map.addListener('click', function (event) { addMarker(event.latLng); }); }
